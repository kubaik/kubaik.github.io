import time
import tkinter as tk

class PomodoroTimer:
    def __init__(self):
        self.root = tk.Tk()
        self.label = tk.Label(self.root, text='25:00', font=('Helvetica', 48))
        self.label.pack()
        self.time_left = 1500  # 25 minutes in seconds
        self.running = False

    def start(self):
        self.running = True
        self.update_timer()

    def update_timer(self):
        if self.running:
            minutes, seconds = divmod(self.time_left, 60)
            self.label.config(text=f'{minutes:02d}:{seconds:02d}')
            self.time_left -= 1
            if self.time_left >= 0:
                self.root.after(1000, self.update_timer)
            else:
                self.running = False
                self.label.config(text='Break Time!')

    def run(self):
        self.start()
        self.root.mainloop()

if __name__ == '__main__':
    timer = PomodoroTimer()
    timer.run()

This code creates a simple GUI-based Pomodoro timer that updates every second.
