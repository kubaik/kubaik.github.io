class ConcurrentQueue {

    void push(T item) {
        queue_.push_back(item);
        cond_.notify_one();
    }

    T pop() {
        while (queue_.empty()) {
            cond_.wait(mutex_);
        }
        T item = queue_.front();
        queue_.pop_front();
        return item;
    }

private:

    std::mutex mutex_;
    std::condition_variable cond_;
